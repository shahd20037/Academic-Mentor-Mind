<!-- Proctoring Logic -->
<script>
    const taskId = {{ task.id }};
    const video = document.getElementById('video');
    const canvas = document.getElementById('canvas');
    const warningList = document.getElementById('warningList');
    let warningCount = 0;

    // 1. Camera & Mic Access
    async function setupMedia() {
        try {
            const stream = await navigator.mediaDevices.getUserMedia({ video: true, audio: true });
            video.srcObject = stream;
            document.getElementById('camStatus').innerText = 'Online';
            document.getElementById('micStatus').innerText = 'Online';
        } catch (err) {
            alert("Camera and Microphone access is required for this exam!");
            window.location.href = "{{ url_for('student_dashboard') }}";
        }
    }

    // 2. Tab Switching Detection
    document.addEventListener('visibilitychange', () => {
        if (document.hidden) {
            logViolation('TAB_SWITCH', 'Student left the exam page');
        }
    });

    window.addEventListener('blur', () => {
        logViolation('WINDOW_BLUR', 'Student switched to another window/app');
    });

    // 3. Prevent Copy/Paste
    document.addEventListener('copy', (e) => {
        e.preventDefault();
        logViolation('COPY_ATTEMPT', 'Student tried to copy text');
    });
    document.addEventListener('paste', (e) => {
        e.preventDefault();
        logViolation('PASTE_ATTEMPT', 'Student tried to paste text');
    });
    document.addEventListener('contextmenu', (e) => e.preventDefault());

    // 4. Logging Violations
    function logViolation(type, details) {
        const li = document.createElement('li');
        li.innerText = `- [${new Date().toLocaleTimeString()}] ${type}: ${details}`;
        warningList.prepend(li);
        
        // Take screenshot
        const context = canvas.getContext('2d');
        canvas.width = video.videoWidth;
        canvas.height = video.videoHeight;
        context.drawImage(video, 0, 0, canvas.width, canvas.height);
        const screenshot = canvas.toDataURL('image/jpeg', 0.5);

        fetch('/api/proctoring/log', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
                task_id: taskId,
                violation_type: type,
                details: details,
                screenshot: screenshot
            })
        });

        warningCount++;
        if (warningCount > 5) {
            alert("Too many violations detected! Your exam may be terminated.");
        }
    }

    // 5. Face Detection Simulation (Simplified for Frontend)
    // In a real app, we'd use face-api.js or MediaPipe here.
    // For now, we'll simulate periodic checks.
    setInterval(() => {
        // Randomly simulate "Looking Away" or "Face Not Found" for demo if needed
        // Or just keep it as a placeholder for where the ML model would go
    }, 5000);

    setupMedia();

    // Timer Logic
    let timeLeft = 3600;
    setInterval(() => {
        if (timeLeft <= 0) return;
        timeLeft--;
        const mins = Math.floor(timeLeft / 60);
        const secs = timeLeft % 60;
        document.getElementById('timer').innerText = `Time Remaining: ${mins}:${secs < 10 ? '0' : ''}${secs}`;
    }, 1000);

    document.getElementById('submitExam').onclick = () => {
        alert("Exam submitted successfully!");
        window.location.href = "{{ url_for('student_dashboard') }}";
    };
</script>
